  group_by(ScoreType) %>%
  summarize(Frequency = n()) %>%
  arrange(desc(Frequency)) %>%
  pull(ScoreType)

# Apply this order to ScoreType in els_long

  mutate(Percentage = Counts / total_counts * 100)
